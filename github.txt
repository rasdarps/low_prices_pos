##Using Github

#adding new project to git
create repo
git init
git add .
git commit -m "new app"
git branch -m "master"
git remote add origin your repo url
git push -u origin "master"


#Switching Branches
git checkout branch_name eg TNC_updates


#Updating a branch on git/ "update" be comment and can be anything
git add .
git commit -m "update"
git push origin -u rictPOS_updates


#To merge
git checkout master
git pull
git merge branch name eg TNC_updates
git commit -m "Merge <feature-branch> into main branch"
git push


#pulling, fetching and merging from the main/master brnch on github
git pull
git fetch 
git merge origin/master
